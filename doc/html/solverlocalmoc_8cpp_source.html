597  }
598  else{

600  _bdryFlux[DOFObj(3*elementID+1,n,g)] = _prob.nxnyq[bi+2];
601  }
602  }
603  }
604  }
605  }
606  }
607 
608  // Edge 2
609  if (vertexNeighbor2 < 0.0) {
610  switch (xedge) {
611  case 0:
612  nx = element.y[v0]-element.y[v2]; ny = element.x[v2]-element.x[v0];
613  break;
614  case 1:
615  nx = element.y[v1]-element.y[v0]; ny = element.x[v0]-element.x[v1];
616  break;
617  case 2:
618  nx = element.y[v2]-element.y[v1]; ny = element.x[v1]-element.x[v2];
619  break;
620  }
621  double norm = sqrt(pow(nx,2)+pow(ny,2));
622  nx = nx/norm;
623  ny = ny/norm;
624  OmegaDotN = sqrt(1.0-pow(_mu[n],2))*cos(_theta[n])*nx +
625  sqrt(1.0-pow(_mu[n],2))*sin(_theta[n])*ny;
626  if (OmegaDotN < 0.0) {
627  if (_prob.globalBC == reflecting) {
628  int np = _getReflectedDirection(n, nx, ny, OmegaDotN);
629  for (int g=0; g<_prob.numGroups; g++)
630  _bdryFlux[DOFObj(3*elementID+2,n,g)] = _solution[ _dofIndex(elementID,np,g) ];
631  }
632  else if (_prob.globalBC == vacuum) {
633  for (int g=0; g<_prob.numGroups; g++)
634  _bdryFlux[DOFObj(3*elementID+2,n,g)] = 0.0;
635  }
636  else if (_prob.globalBC == source) {
637  for (int bi=0; bi<_prob.nxnyq.size(); bi+=3) {
638  if (std::abs(nx-_prob.nxnyq[bi]) < 1.0e-8 && std::abs(ny-_prob.nxnyq[bi+1]) < 1.0e-8) {
639  if (_prob.nxnyq[bi+2]<0.0) {
